	\(u\equiv 0\).
	\item observe que ocurre en el inciso anterior, pero con la
	condición 
	\begin{displaymath}
		\norm{u_{\infty}}_{L^{1}(B(0,r))} < r^{n-1}.
	\end{displaymath}	
\end{enumerate}
